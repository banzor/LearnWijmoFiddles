onload = function() {

	// create several InputDate controls and assign them formats
  var parent = document.getElementById('inputDates');
  var formats = 'd|D|M|d/M/yy|MMMM dd, yyyy'.split('|');
	var template = '<div>' +
  	'<label for={id}>{fmt}: </label>' +
    '<input id={id}>' +
    '</div>';
  for (var i = 0; i < formats.length; i++) {
  	var fmt = formats[i],
    		id = 'theInputDateF' + i;
  	var html = wijmo.format(template, {
    	fmt: fmt,
      id: id
		});
  	parent.appendChild(wijmo.createElement(html));
  	var theDate = new wijmo.input.InputDate('#' + id, {
  		format: fmt
    });
  }
}
