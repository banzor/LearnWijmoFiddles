onload = function() {

	// create the gauge
	var theGauge = new wijmo.gauge.LinearGauge('#theGauge', {
  	isReadOnly: false,
  	min: 0,
    max: 100,
    step: 5,
    value: 75,
    valueChanged: function(s, e) {
    	value.value = s.value;
    },
    showRanges: false,
    ranges: [
    	{ min: 0, max: 33, color: 'red' },
    	{ min: 33, max: 66, color: 'yellow' },
    	{ min: 66, max: 100, color: 'green' },
    ]
  });
  setDirection('Right');
  
  // change properties
  var value = new wijmo.input.InputNumber('#value', {
  	min: theGauge.min,
    max: theGauge.max,
    step: theGauge.step,
    value: theGauge.value,
    valueChanged: function(s, e) {
    	theGauge.value = s.value;
    }
  });
  var direction = new wijmo.input.ComboBox('#direction', {
    itemsSource: 'Left,Right,Up,Down'.split(','),
    text: 'Right',
    textChanged: function(s, e) {
    	setDirection(s.text);
    }
  });

	function setDirection(dir) {
		theGauge.direction = dir;
      var g = theGauge.hostElement.style;
      switch (dir) {
      	case 'Left':
      	case 'Right':
        	g.height = '2em';
          g.width = '100%';
        	break;
      	case 'Up':
      	case 'Down':
        	g.height = '100%';
          g.width = '2em';
        	break;
      }
	}
}
