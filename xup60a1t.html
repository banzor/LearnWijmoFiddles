onload = function () {

  // generate some random data
  var countries = 'US,Germany,UK,Japan,Italy,Greece'.split(','),
      data = [];
  for (var i = 0; i < countries.length; i++) {
    data.push({
    	id: i,
      country: countries[i],
      downloads: Math.round(Math.random() * 20000),
      sales: Math.random() * 10000,
      expenses: Math.random() * 5000
    });
  }

	// toggle row visibility
  var hideRows = false;
  document.getElementById('btnToggleVis').addEventListener('click', function(e) {
  	hideRows = !hideRows;
    theGrid.collectionView.refresh(); // force row re-load
  });
  
	// grid with custom columns
  var theGrid = new wijmo.grid.FlexGrid('#theGrid', {
    itemsSource: data,
    loadedRows: function() {
    	if (hideRows) {
      	theGrid.rows.forEach(function(row, index) {
        	row.visible = index % 2 == 0;
        });
      }
    }
  });
}

  <h1>
    Row Properties</h1>
  <p>
    The <b>wijmo.grid.Row</b> class has 
    <a href="https://wijmo.com/5/docs/topic/wijmo.grid.Row.Class.html" target="_blank">
      almost 20 properties 
    </a> you can use to configure each rows's appearance and behavior.</p>
  <p>
    Note that when grids refresh, the scrollable rows are re-created 
    (so they can be bound to the new data). Because of this, you
    should apply normally apply row properties in response to the
    <b>loadedRows</b> event, which fires after the rows have been
    created.</p>
  <p>
    Most rows are instances of the <b>Row</b> class, but grouped grids
    may also contain <b>GroupRow</b> objects, which extend the regular
    <b>Row</b> class.</p>
  <p>
    The most important properties in the <b>Row</b> class are:</p>
  <ul>
    <li>
      <b>dataItem</b>:
      Contains a reference to the data item that is bound to the row.
      This property is set by the grid when it creates bound rows, and
      is often used in <b>formatItem</b> event handlers.
      In <b>GroupRow</b> objects, the <b>dataItem</b> property 
      contains a reference to the row's <b>Group</b> rather than
      to a regular data item.</li>
    <li>
      <b>isReadOnly</b>:
      Gets or sets whether the row is editable. By default, <b>Row</b>
      instances are editable and <b>GroupRow</b> instances are not.</li>
    <li>
      <b>isSelected</b>:
      This property is important when the grid's <b>selectionMode</b>
      property is set to <b>ListBox</b>. In this case, users may select
      non-contiguous sets of rows, and the <b>isSelected</b> property
      allows you to get or set the selected state of individual rows.</li>
    <li>
      <b>visible</b>:
      Gets or sets whether the row is visible. You can use this property
      to conditionally hide rows.</li>
    <li>
      <b>isVisible</b>:
      Gets a value that indicates whether the row is currently visible.
      Even if the <b>visible</b> property is set to true, rows may
      be hidden because they are part of collapsed groups.</li>
  </ul>
      
  <p>
    For example, click the button below to hide or show every other
    row on the grid:</p>    
  <button id="btnToggleVis" class="btn btn-default">
    Toggle Row Visibility
  </button>
