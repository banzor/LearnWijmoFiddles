onload = function() {

	// create the chart
  var theChart = new wijmo.chart.FlexChart('#theChart', {
    axisY: {
    	format: 'n0,',
      title: 'US$ (thousands)',
      itemFormatter: function(engine, label) {
    		if (label.val >= 4000){
        	label.cls = 'large-value';
    	 	}
    		return label;
			}
    },
  	itemsSource: getData(),
    bindingX: 'country',
    series: [
    	{ binding: 'sales', name: 'Sales' },
      { binding: 'expenses', name: 'Expenses' },
    ]
  });

	// create some random data
  function getData() {
	  var countries = 'US,Canada,Mexico,Germany,UK,France,Italy,Japan,Korea,China'.split(','),
	    data = [];
	  for (var i = 0; i < countries.length; i++) {
	    data.push({
	      country: countries[i],
	      sales: 1000 + Math.random() * 10000,
	      expenses: 1000 + Math.random() * 5000,
	      downloads: Math.round(Math.random() * 20000),
	    });
	  }
  	return data;
  }

}

  <ol>
    <li>
      <b>engine</b>: The IRenderEngine object used for rendering the labels.</li>
    <li>
      <b>label</b>: An object that represents the label and has these properties:
        <ul>
          <li>
            <b>value</b>: The value that the label represents.</li>
          <li>
            <b>text</b>: The text content of the label 
            (usually the formatted value).</li>
          <li>
            <b>pos</b>: The position where the label will be rendered, 
            in control coordinates.</li>
          <li>
            <b>cls</b>: A CSS class to be applied to the label element.</li>
        </ul>
      </li>
  </ol>
