onload = function() {

  // create some random data
  var countries = 'US,Germany,UK,Japan,Italy,Greece'.split(','),
    data = [];
  for (var i = 0; i < 200; i++) {
    data.push({
      id: i,
      country: countries[i % countries.length],
      active: i % 5 != 0,
      downloads: Math.round(Math.random() * 200000),
      sales: Math.random() * 100000,
      expenses: Math.random() * 50000,
      v1: Math.random(),
      v2: Math.random(),
      v3: Math.random(),
      v4: Math.random(),
      v5: Math.random()
    });
  }

  // default grid
  var theGrid = new wijmo.grid.FlexGrid('#theGrid', {
    itemsSource: data,
    allowDragging: 'Both',
    
    // re-order items in the data source after dragging a row
    draggedRow: function(s, e) {
    	var sp = s.scrollPosition;
    	s.collectionView.sourceCollection = s.rows.map(function(row) {
      	return row.dataItem; 
			});
      s.scrollPosition = sp;
    }
  });
}
