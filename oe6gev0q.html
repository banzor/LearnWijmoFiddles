onload = function() {

	// create gauges
	var theRadialGauge = new wijmo.gauge.RadialGauge('#theRadialGauge', {
  	min: 0,
    max: 100,
    value: 75,
    isReadOnly: false,
    step: 10,
    ranges: [
    	{ name: 'low', min: 0, max: 25, color: 'rgba(255,0,0,.4)' },
    	{ name: 'med', min: 25, max: 50, color: 'rgba(255,255,0,.4)' },
    	{ name: 'high', min: 50, max: 75, color: 'rgba(255,125,0,.4)' },
    	{ name: 'top', min: 75, max: 100, color: 'rgba(0,255,0,.4)' },
    ],
    valueChanged: function(s, e) {
    	theLinearGauge.value = s.value;
    }
  });
  theRadialGauge.pointer.thickness = .2;
  theRadialGauge.pointer.color = 'black';
  
	var theLinearGauge = new wijmo.gauge.LinearGauge('#theLinearGauge', {
  	min: 0,
    max: 100,
    value: 75,
    isReadOnly: false,
    step: 10,
    ranges: [
    	{ name: 'low', min: 0, max: 25, color: 'rgba(255,0,0,.4)' },
    	{ name: 'med', min: 25, max: 50, color: 'rgba(255,255,0,.4)' },
    	{ name: 'high', min: 50, max: 75, color: 'rgba(255,125,0,.4)' },
    	{ name: 'top', min: 75, max: 100, color: 'rgba(0,255,0,.4)' },
    ],
    valueChanged: function(s, e) {
    	theRadialGauge.value = s.value;
    }
  });
  theLinearGauge.pointer.thickness = .4;
  theLinearGauge.pointer.color = 'black';

	document.getElementById('showRanges').addEventListener('click', function(e) {
  	if (e.target.checked) {
  		theRadialGauge.showRanges = theLinearGauge.showRanges = true;
			theRadialGauge.pointer.thickness = .2;
		  theLinearGauge.pointer.thickness = .4;
		} else {
  		theRadialGauge.showRanges = theLinearGauge.showRanges = false;
			theRadialGauge.pointer.thickness = 1;
		  theLinearGauge.pointer.thickness = 1;
    }
	});

}
<div class="container">
    
  <h1>
    Ranges</h1>
  <p>
    Wijmo gauges are composed of <b>Range</b> objects.
    Range objects have "min" and "max" properties the define
    the extent of the range, as well as "thickness" and "color"
    properties that define the appearance of the range.
  <p>
    Every gauge has at least two ranges, the "face" and
    the "pointer":</p>
  <ul>
    <li>
      <b>face</b>:
        Represents the gauge background. The "min" and "max" properties
        of the face range correspond to the "min" and "max" properties
        of the gauge, and limit the values that the gauge can display.</li>
    <li>
      <b>pointer</b>:
        Indicates the gauge's current value. The "max" property of
        the pointer range corresponds to the "value" property of 
        the gauge.</li>
  </ul>
  <p>
    In addition to these two special ranges, gauges may have any number
    of additional ranges added to their "ranges" collection. These 
    additional ranges can be used in two ways:</p>
  <p>
    By default, extra ranges appear as part of the gauge background.
    This way you can show 'zones' within the gauge, like 'good',
    'average', and 'bad' for example.
    If you set the gauge's "showRanges" property to false, the
    additional ranges are not shown. Instead, they are used to
    automatically set the color of the "pointer" range based on
		the current gauge value:</p>
  <label>
    showRanges
    <input id="showRanges" type="checkbox" checked="checked">
  </label>
  <div id="theRadialGauge"></div>
  <div id="theLinearGauge"></div>
</div>
