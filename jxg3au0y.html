onload = function() {

	// create the gauge
  var theGauge = new wijmo.gauge.RadialGauge('#theGauge', {
  	isReadOnly: false,
    hasShadow: false,
    showText: 'All',
    startAngle: -30,
    sweepAngle: 240,
    value: 50,
  });
  theGauge.pointer.thickness = .15;
  theGauge.thumbSize = 20;

	var theColor = new wijmo.input.InputColor('#theColor', {
  	value: theGauge.face.color,
		valueChanged: function(s, e) {
			theGauge.face.color = s.value;
		}	
	});
	var theThickness = new wijmo.input.InputNumber('#theThickness', {
  	min: 0,
  	max: 1,
    step: .1,
    format: 'n2',
    value: theGauge.face.thickness,
		valueChanged: function(s, e) {
			theGauge.face.thickness = s.value;
		}	
	});

}
