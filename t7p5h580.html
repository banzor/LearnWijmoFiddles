onload = function() {

	// create the chart
  var theChart = new wijmo.chart.FlexChart('#theChart', {
  	itemsSource: getData(),
    bindingX: 'country',
    series: [
    	{ binding: 'sales', name: 'Sales' },
      { binding: 'expenses', name: 'Expenses' },
      { binding: 'downloads', name: 'Downloads', chartType: 'LineSymbols' },
    ],
    axisY: {
    	format: 'n0,',
      title: 'Sales/Expenses (US$ k)',
      axisLine: true
    },
  });
  
  // create and apply extra Y axis for 'Downloads' series
  var axisY2 = new wijmo.chart.Axis();
  axisY2.position = 'Right';
  axisY2.title = 'Downloads (k)';
  axisY2.format = 'n0,';
  axisY2.min = 0;
  axisY2.axisLine = true;
  getSeries('downloads').axisY = axisY2;

	// toggle extra axis
  document.getElementById('secondaryAxis').addEventListener('click', function(e) {
  	getSeries('downloads').axisY = e.target.checked ? axisY2 : null;
  });
  
  // get a series by its binding
  function getSeries(binding) {
  	var s = theChart.series;
  	for (var i = 0; i < s.length; i++) {
    	if (s[i].binding == binding) {
    		return s[i];
    	}
    }
    return null;
  }
  
	// create some random data
  function getData() {
	  var countries = 'US,Canada,Mexico,Germany,UK,France,Italy,Japan,Korea,China'.split(','),
	    data = [];
	  for (var i = 0; i < countries.length; i++) {
	    data.push({
	      country: countries[i],
	      sales: 1000 + Math.random() * 10000,
	      expenses: 1000 + Math.random() * 5000,
	      downloads: Math.round(Math.random() * 100000),
	    });
	  }
  	return data;
  }

}

<div class="container">

  <h1>
    Extra Axes
  </h1>

  <p>
    Most charts have two axes, X and Y. This works well as long
    as all the data on the chart has the same nature and can 
    share the same scale.</p>
  <p>
    But some charts contain series that show different types of
    data, with different units and scales. The chart below has
    two series that represent amounts (sales and expenses) and
    one that represents quantities (downloads).</p>
  <p>
    Plotting all the series against a single Y axis squeezes 
    the first two series against the bottom of the chart.</p>
  <p>
    The easiest way to solve this problem and still using a 
    single chart is to create a secondary Y axis and assign
    it to the <b>axisY</b> property of the "Downloads" series:</p>

  <label for="secondaryAxis">Secondary Y Axis</label>
  <input id="secondaryAxis" type="checkbox" checked="checked">
  
  <div id="theChart"></div>
</div>
