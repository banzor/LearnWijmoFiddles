onload = function() {

  // create some random data
  var countries = 'US,Germany,UK,Japan,Italy,Greece,Spain,Portugal,Australia'.split(',');
  var data = [];
  for (var i = 0; i < countries.length; i++) {
    data.push({
      country: countries[i],
      downloads: Math.round(Math.random() * 20000),
      sales: Math.random() * 10000,
      expenses: Math.random() * 5000
    });
  }

  // create a CollectionView
  var view = new wijmo.collections.CollectionView(data, {
		sortDescriptions: ['country'],
    trackChanges: true
  });

  // create grid for editing
  var theGrid = new wijmo.grid.FlexGrid('#theGrid', {
    itemsSource: view,
    allowAddNew: true,
    allowDelete: true
  });
  
  // create grids to show changes
  var edited = new wijmo.grid.FlexGrid('#edited', {
    itemsSource: view.itemsEdited,
    isReadOnly: true
  });
  var added = new wijmo.grid.FlexGrid('#added', {
    itemsSource: view.itemsAdded,
    isReadOnly: true
  });
  var removed = new wijmo.grid.FlexGrid('#removed', {
    itemsSource: view.itemsRemoved,
    isReadOnly: true
  });
  
}

    Web applications often use a pattern of downloading

  <p>
    The <b>CollectionView</b> can help by keeping track
    of items that have been added, removed, or modified.</p>
  <p>
    To use this feature, set the 
    <a href="https://wijmo.com/5/docs/topic/wijmo.collections.CollectionView.Class.html#trackChanges" target="_blank">trackChanges</a>
    property to true. Once you do that, the <b>CollectionView</b>
    will add items to its <b>itemsAdded</b>, <b>itemsRemoved</b>,
    and <b>itemsEdited</b> collections. When you are ready, send
    the changes to the server and call the <b>clearChanges</b>
    method to reset the change-tracking collections.</p>
  <p>
