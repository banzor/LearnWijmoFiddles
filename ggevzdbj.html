onload = function() {

  // create some random data
  var countries = 'US,Germany,UK,Japan,Italy,Greece,Spain,Portugal,Australia'.split(',');
  var data = [];
  for (var i = 0; i < countries.length; i++) {
    data.push({
      country: countries[i],
      downloads: Math.round(Math.random() * 20000),
      sales: Math.random() * 10000,
      expenses: Math.random() * 5000
    });
  }

  // create a CollectionView
  var view = new wijmo.collections.CollectionView(data, {
		sortDescriptions: ['country']    
  });

  // create grid for editing
  var theGrid = new wijmo.grid.FlexGrid('#theGrid', {
    itemsSource: view,
    allowAddNew: true,
    allowDelete: true
  });
}

  <h1>
    CollectionView Adding and Removing Items
  </h1>
  <p>
    The <b>CollectionView</b> provides support for adding
    items with the methods <b>addNew</b>, <b>commitNew</b>,
    and <b>cancelNew</b>. Removing items is done with the
    <b>remove</b> method.</p>
  <p>
    The <b>addNew</b> method adds an empty item to the 
    collection and returns a reference to the new item.
    The caller can use this return value to initialize
    the new item. Alternatively, you can provide a 
    <b>CollectionView.newItemCreator</b> function to 
    create and initialize new items.</p>
  <p>
    The <b>addNew</b> method also puts the collection in
    'add' mode, suspending sorting and filtering to keep
    the new item in place until it is committed.</p>
  <p>
    The <b>commitNew</b> method causes the collection
    to exit 'add mode' and refreshes the collection
    so sorting and filtering are restored.</p>
  <p>
    The <b>cancelNew</b> method removes the new item from
    the collection and exits 'add mode'.</p>
  <p>
    You can see how this works by adding items to the 
    grid below.</p>
  <p>
    To add an item, move the selection to the last 
    row, the "New Item Template" (it has an asterisk on 
    the row header). Edit the new item as usual, and
    when you are done press Enter or move the selection 
    to a different row to commit the new row.</p>
  <p>
    If you press the Escape key while editing the new
    row, the addition will be canceled and the new row will
    be removed from the collection.</p>
  <p>
    To remove items, select an entire row by clicking 
    the row header, then press the Delete key. The grid
    will call the collection's <b>remove</b> method
    and the item will be removed from the collection.</p>
