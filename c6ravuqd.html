onload = function() {

	var theLinearGauge = new wijmo.gauge.LinearGauge('#theLinearGauge', {
      min: 0,
      max: 255,
      step: 20,
      isReadOnly: false,
      thumbSize: 12,
      hasShadow: false
	});
  theLinearGauge.face.thickness = .25;
  theLinearGauge.pointer.thickness = .25;

	var theRadialGauge = new wijmo.gauge.RadialGauge('#theRadialGauge', {
      min: 0,
      max: 255,
      step: 10,
      showText: 'Value',
      isReadOnly: false,
      thumbSize: 7,
	});
  theRadialGauge.face.thickness = .07;
  theRadialGauge.pointer.thickness = .07;
}
