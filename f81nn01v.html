onload = function() {

	// create a chart
  var theChart = new wijmo.chart.FlexChart('#theChart', {
  	itemsSource: getData(),
    bindingX: 'country',
    series: [
    	{ binding: 'sales', name: 'Sales' },
      { binding: 'expenses', name: 'Expenses' },
      {	binding: 'downloads', name: 'Downloads' }
    ]
  });
  
	// customize chart titles
  var header = new wijmo.input.ComboBox('#header', {
  	textChanged: function(s, e) {
    	theChart.header = s.text;
		}
  });
  var footer = new wijmo.input.ComboBox('#footer', {
  	textChanged: function(s, e) {
    	theChart.footer = s.text;
		}
  });
  var xTitle = new wijmo.input.ComboBox('#xTitle', {
  	textChanged: function(s, e) {
    	theChart.axisX.title = s.text;
		}
  });
  var yTitle = new wijmo.input.ComboBox('#yTitle', {
  	textChanged: function(s, e) {
    	theChart.axisY.title = s.text;
		}
  });

	// initialize titles
  header.text = 'My Great Chart'
  footer.text = 'powered by Wijmo\'s FlexChart';
  xTitle.text = 'country';
  yTitle.text = 'values/units';

	// create some random data
  function getData() {
	  var countries = 'US,Germany,UK,Japan,Italy,Greece'.split(','),
	    data = [];
	  for (var i = 0; i < countries.length; i++) {
	    data.push({
	      country: countries[i],
	      sales: Math.random() * 10000,
	      expenses: Math.random() * 5000,
	      downloads: Math.round(Math.random() * 20000),
	    });
	  }
  	return data;
  }

}
