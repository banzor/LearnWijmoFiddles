onload = function() {

  // create the chart
  var theChart = new wijmo.chart.FlexChart('#theChart', {
  	header: 'Phase II Test Results',
    chartType: 'Scatter',
    axisY: {
      axisLine: true
    },
    series: [
    	{
      	name: 'Experiment 1', 
        itemsSource: getData(50, 0, 3),
        bindingX: 'x',
        binding: 'y' },
    	{
      	name: 'Experiment 2',
        itemsSource: getData(40, 100, 12),
        bindingX: 'x', 
        binding: 'y' },
    	{
      	name: 'Experiment 3',
        itemsSource: getData(30, -100, 24),
        bindingX: 'x', 
        binding: 'y' },
    ]
  });

	// save chart image to file
  document.getElementById('saveButtons').addEventListener('click', function(e) {
  	if (e.target instanceof HTMLButtonElement) {
    	theChart.saveImageToFile('FlexChart.' + e.target.textContent);
    }
  })
  
	// data sources
  function getData(cnt, a, b) {
  	var arr = [],
    	  x = -5 * cnt / 2;
    for (i = 0; i < cnt; i++) {
    	var rnd = Math.random() * cnt - cnt / 2;
    	arr.push({
      	x: x,
        y: a + x * (b + rnd) + rnd
      });
      x += .5 + Math.random() * 10;
    }
    return arr;
  }
}
