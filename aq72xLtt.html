onload = function() {

	// create the chart for showing amounts
  var theChartAmounts = new wijmo.chart.FlexChart('#theChartAmounts', {
  	itemsSource: getData(),
    header: 'Sales, Expenses, and Downloads',
    bindingX: 'country',
    axisX: { position: 'None' },
    series: [
    	{ binding: 'sales', name: 'Sales' },
      { binding: 'expenses', name: 'Expenses' },
    ],
    plotMargin: 'NaN 120 10 60' // top, right, bottom, left
  });

	// create the chart for showing quantities
  var theChartQty = new wijmo.chart.FlexChart('#theChartQty', {
  	itemsSource: getData(),
    bindingX: 'country',
    series: [
      { binding: 'downloads', name: 'Downloads', chartType: 'LineSymbols' },
    ],
    plotMargin: '10 120 NaN 60' // top, right, bottom, left
  });

	// create some random data
  function getData() {
	  var countries = 'US,Germany,UK,Japan,Italy,Greece'.split(','),
	    data = [];
	  for (var i = 0; i < countries.length; i++) {
	    data.push({
	      country: countries[i],
	      sales: Math.random() * 10000,
	      expenses: Math.random() * 5000,
	      downloads: Math.round(Math.random() * 20000),
	    });
	  }
  	return data;
  }

}
