onload = function() {

  var theGrid = new wijmo.grid.FlexGrid('#theGrid', {
    showAlternatingRows: false,
    showMarquee: true,
  	itemsSource: getData(),
  });

	// toggle marquee styling
	document.getElementById('showMarquee').addEventListener('click', function(e) {
  	theGrid.showMarquee = e.target.checked;
    theGrid.focus();
  });
	document.getElementById('customMarquee').addEventListener('click', function(e) {
  	wijmo.toggleClass(theGrid.hostElement, 'customMarquee', e.target.checked);
    theGrid.invalidate();
    theGrid.focus();
  });
  document.getElementById('highlightActive').addEventListener('click', function(e) {
  	wijmo.toggleClass(theGrid.hostElement, 'highlightActive', e.target.checked);
    theGrid.focus();
  });

	// create some random data
  function getData() {
	  var countries = 'US,UK,Germany,France,Italy,Japan,China'.split(','),
  	  data = [];
  	for (var i = 0; i < countries.length; i++) {
	    data.push({
	      country: countries[i],
	      downloads: Math.round(Math.random() * 20000),
	      sales: Math.random() * 10000,
	      expenses: Math.random() * 5000
	    });
	  }
		return data;
  }
}
