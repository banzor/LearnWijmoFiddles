onload = function() {

	// a regular input number
  var theNumber = new wijmo.input.InputNumber('#theNumber', {
  	valueChanged: function(s, e) {
    	document.getElementById('theNumberOutput').textContent = s.value;
    }
  });

	// you can clear this one
  var theNumberNotRequired = new wijmo.input.InputNumber('#theNumberNotRequired', {
  	isRequired: false,
    value: null,
    min: 0,
    max: 100,
    placeholder: 'Your age (optional)'
  });
}
