onload = function() {

  // initialize pivot engine
  var svcUrl = 'https://demos.componentone.com/ASPNET/c1webapi/4.0.20173.114/api/dataengine/';
  var ngCube = new wijmo.olap.PivotEngine({
  
  	// connect to ComponentOne Data Service
    itemsSource:  svcUrl + 'complex10',
    
    // build default view
    valueFields: ['Sales', 'Downloads'],
    rowFields: ['Country', 'Product']
  });
  
  // show pivot panel
  var pivotPanel = new wijmo.olap.PivotPanel('#pivotPanel', {
    itemsSource: ngCube
  });

  // show summary
  var pivotGrid = new wijmo.olap.PivotGrid('#pivotGrid', {
    itemsSource: ngCube
  });

	// count fields (and subfields)
	function countFields(arr) {
  	cnt = 0;
    arr.forEach(function(field) {
      if (wijmo.isArray(field.subFields)) {
      	cnt += countFields(field.subFields)
      } else {
	    	cnt++;
      }
    });
		return cnt;    
  }
}
