onload = function() {

	// create gauges
	var theRadialGauge = new wijmo.gauge.RadialGauge('#theRadialGauge', {
  	min: 0,
    max: 100,
    value: 75,
    showText: 'None',
    valueChanged: function(s, e) {
      valueCtl.value = s.value;
    }
  });
	var theLinearGauge = new wijmo.gauge.LinearGauge('#theLinearGauge', {
  	min: 0,
    max: 100,
    value: 75,
    showText: 'None',
    valueChanged: function(s, e) {
      valueCtl.value = s.value;
    }
  });
	var theBulletGraph = new wijmo.gauge.BulletGraph('#theBulletGraph', {
  	min: 0,
    max: 100,
    value: 75,
    target: 60,
    bad: 50,
    good: 70,
    showText: 'None',
    valueChanged: function(s, e) {
      valueCtl.value = s.value;
    }
  });
	function getGaugeProp(prop) {
  	return theRadialGauge[prop];
  }
	function setGaugeProp(prop, value) {
  	theRadialGauge[prop] = value;
  	theLinearGauge[prop] = value;
  	theBulletGraph[prop] = value;
  }


	// create controls to change gauge properties
  var minCtl = new wijmo.input.InputNumber('#min', {
  	step: 10,
    value: getGaugeProp('min'),
    valueChanged: function(s, e) {
    	setGaugeProp('min', s.value)
    }
  });
  var maxCtl = new wijmo.input.InputNumber('#max', {
  	step: 10,
    value: getGaugeProp('max'),
    valueChanged: function(s, e) {
    	setGaugeProp('max', s.value)
    }
  });
  var valueCtl = new wijmo.input.InputNumber('#value', {
  	step: 10,
    value: getGaugeProp('value'),
    valueChanged: function(s, e) {
    	setGaugeProp('value', s.value)
    }
  });
  var showTextCtl = new wijmo.input.ComboBox('#showText', {
  	itemsSource: 'None,Value,MinMax,All'.split(','),
		text: getGaugeProp('showText'),
		textChanged: function(s, e) {
    	setGaugeProp('showText', s.text);
    }
  });
  document.getElementById('isReadOnly').addEventListener('click', function(e) {
  	setGaugeProp('isReadOnly', e.target.checked);
  });
  var stepCtl = new wijmo.input.InputNumber('#step', {
  	step: 1,
    value: getGaugeProp('step'),
    valueChanged: function(s, e) {
    	setGaugeProp('step', s.value)
    }
  });
	document.getElementById('isAnimated').addEventListener('click', function(e) {
  	setGaugeProp('isAnimated', e.target.checked);
  });
}
