onload = function() {

	// color being edited
  var theColor = new wijmo.Color('grey');
  var thePanel = document.getElementById('thePanel');
  updateColor();
  function updateColor() {
    thePanel.style.background = theColor;
  }

	// create the gauges
  var red = new wijmo.gauge.LinearGauge('#red', {
    value: theColor.r,
    valueChanged: function(s, e) {
    	theColor.r = s.value;
      updateColor();
    }
  });
  initGauge(red, 'red');
  
  var green = new wijmo.gauge.LinearGauge('#green', {
    value: theColor.g,
    valueChanged: function(s, e) {
    	theColor.g = s.value;
      updateColor();
    }
  });
  initGauge(green, 'green');
  
  var blue = new wijmo.gauge.LinearGauge('#blue', {
    value: theColor.b,
    valueChanged: function(s, e) {
    	theColor.b = s.value;
      updateColor();
    }
  });
  initGauge(blue, 'blue');
  
  function initGauge(g, color) {
		g.isReadOnly = false,
  	g.thumbSize = 10,
  	g.hasShadow = false,
  	g.min = 0;
  	g.max = 255;
  	g.step = 5,
    g.showTicks = true;
    g.tickSpacing = 25;
  	g.face.thickness = .2;
  	g.pointer.thickness = .3;
  	g.pointer.color = color;
  	g.hostElement.style.color = color;
	}  
 
}
