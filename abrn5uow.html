onload = function() {

	// create the chart
  var theChart = new wijmo.chart.FlexChart('#theChart', {
  	itemsSource: getData(),
    bindingX: 'country',
    series: [
    	{ binding: 'sales', name: 'Sales' }
    ],
    axisY: {
    	format: 'n0,',
      title: 'US$ (thousands)'
    },
    legend: {
    	position: 'None'
    },
  });

	// toggle the custom range
  var warning = document.getElementById('warning');
  document.getElementById('customRange').addEventListener('click', function(e) {
  	if (e.target.checked) {
	  	theChart.axisY.min = 150000;
      theChart.axisY.max = 160000;
      warning.style.display = '';
		} else {
    	theChart.axisY.min = theChart.axisY.max = null;
      warning.style.display = 'none';
    }
  });

	// create some random data
  function getData() {
	  var countries = 'US,Canada,Mexico,Germany,UK,France,Italy,Japan,Korea,China'.split(','),
	    data = [];
	  for (var i = 0; i < countries.length; i++) {
	    data.push({
	      country: countries[i],
	      sales: 150000 + Math.random() * 10000,
	      expenses: 10000 + Math.random() * 5000,
	    });
	  }
  	return data;
  }

}
