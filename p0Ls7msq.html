onload = function() {

	// create the tree
  var tree = new wijmo.nav.TreeView('#theTree', {
  	itemsSource: getData(),
		displayMemberPath: 'header',
    childItemsPath: 'items',
    itemClicked: function(s, e) {
    	var msg = document.getElementById('msg');
      msg.innerHTML = wijmo.format('Navigating to <b>** {header} **</b>',
      	s.selectedItem);
    }
	});
