onload = function() {

	// create the tree
  var tree = new wijmo.nav.TreeView('#theTree', {
  	itemsSource: getData(),
		displayMemberPath: 'header',
    childItemsPath: 'items'
	});
  
  // handle checkboxes
  document.getElementById('customCSS').addEventListener('click', function(e) {
  	wijmo.toggleClass(tree.hostElement, 'custom-tree', e.target.checked);
  });
  
  // get the tree data
  function getData() {
		return [
    	{ header: 'Electronics', img: 'resources/electronics.png', items: [
