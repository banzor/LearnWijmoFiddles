onload = function() {

  // create the chart
  var theChart = new wijmo.chart.FlexChart('#theChart', {
    chartType: 'Scatter',
    axisY: {
      axisLine: true,
    },
    legend: {
    	position: 'None'
    },
    series: [
    	{
      	name: 'Experiment 1', 
        itemsSource: getData(50, 0, 3),
        bindingX: 'x',
        binding: 'y' },
    	{
      	name: 'Experiment 2',
        itemsSource: getData(40, 100, 12),
        bindingX: 'x', 
        binding: 'y' },
    	{
      	name: 'Experiment 3',
        itemsSource: getData(30, -100, 24),
        bindingX: 'x', 
        binding: 'y' },
    ]
  });
  
  // change axis postion and origin
  var xPos = new wijmo.input.ComboBox('#xPosition', {
  	itemsSource: 'None,Top,Bottom,Auto'.split(','),
    textChanged: function(s, e) {
    	theChart.axisX.position = s.text;
    },
    text: 'Bottom'
  });
  document.getElementById('xOrigin').addEventListener('click', function(e) {
  	theChart.axisX.origin = e.target.checked ? 0 : null;
  });

	var yPos = new wijmo.input.ComboBox('#yPosition', {
  	itemsSource: 'None,Left,Right,Auto'.split(','),
    textChanged: function(s, e) {
    	theChart.axisY.position = s.text;
    },
    text: 'Left'
  });
  document.getElementById('yOrigin').addEventListener('click', function(e) {
  	theChart.axisY.origin = e.target.checked ? 0 : null;
  });
  
	// data sources
  function getData(cnt, a, b) {
  	var arr = [],
    	  x = -5 * cnt / 2;
    for (i = 0; i < cnt; i++) {
    	var rnd = Math.random() * cnt - cnt / 2;
    	arr.push({
      	x: x,
        y: a + x * (b + rnd) + rnd
      });
      x += .5 + Math.random() * 10;
    }
    return arr;
  }
}
