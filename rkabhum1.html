onload = function() {

	// update radial gauge when value changes
  var theInputRange = document.getElementById('theInputRange');
	theInputRange.addEventListener('input', function(e) {
  	setValue(e.target.value * 1);
  });
  
	// create a LinearGauge with default settings
	var theLinearGauge = new wijmo.gauge.LinearGauge('#theLinearGauge', {
      min: 0,
      max: 255,
      step: 10,
      isReadOnly: false,
      valueChanged: function(s, e) {
		  	setValue(s.value);
      }
	});

	// create a LinearGauge with some styling
	var theLinearGaugeCSS = new wijmo.gauge.LinearGauge('#theLinearGaugeCSS', {
      min: 0,
      max: 255,
      step: 10,
      isReadOnly: false,
      thumbSize: 12,
      hasShadow: false,
      valueChanged: function(s, e) {
		  	setValue(s.value);
      }
	});
  theLinearGaugeCSS.face.thickness = .25;
  theLinearGaugeCSS.pointer.thickness = .25;
  theLinearGaugeCSS.pointer.color = 'red';
  
  // update all controls when the value changes
  var theValue = document.getElementById('theValue');
  setValue(50);
  function setValue(value) {
  	theValue.textContent = value;
  	theInputRange.value = value;
    theLinearGauge.value = value;
		theLinearGaugeCSS.value = value;
  }
  
  // create a group of vertical gauges
  var parent = document.getElementById('gauges');
  for (var i = 0; i < 8; i++) {
  	var gauge = new wijmo.gauge.LinearGauge(document.createElement('div'), {
    	min: 0,
      max: 10,
      value: Math.round(Math.random() * 10),
      isReadOnly: false,
      thumbSize: 20,
			showTicks: true,
      hasShadow: false,
      direction: 'Up',
      showText: 'Value'
    });
  	gauge.face.thickness = .1;
  	gauge.pointer.thickness = .25;
    gauge.hostElement.classList.add('vertical-gauge', 'custom-gauge');
    gauges.appendChild(gauge.hostElement);
  }
}
