onload = function() {

	// update radial gauge when value changes
  var theInputRange = document.getElementById('theInputRange');
	theInputRange.addEventListener('input', function(e) {
  	setValue(e.target.value * 1);
  });
  
	// create a RadialGauge with default settings
	var theRadialGauge = new wijmo.gauge.RadialGauge('#theRadialGauge', {
      min: 0,
      max: 255,
      step: 10,
      showText: 'Value',
      isReadOnly: false,
      valueChanged: function(s, e) {
		  	setValue(s.value);
      }
	});

	// create a RadialGauge with some styling
	var theRadialGaugeCSS = new wijmo.gauge.RadialGauge('#theRadialGaugeCSS', {
      min: 0,
      max: 255,
      step: 10,
      showText: 'Value',
      isReadOnly: false,
      startAngle: -90,
      sweepAngle: 180,
      thumbSize: 8,
      valueChanged: function(s, e) {
		  	setValue(s.value);
      }
	});
  theRadialGaugeCSS.face.thickness = .15;
  theRadialGaugeCSS.pointer.thickness = .15;
  theRadialGaugeCSS.pointer.color = 'red';
  
  // update all controls when the value changes
  var theValue = document.getElementById('theValue');
  setValue(50);
  function setValue(value) {
    theValue.textContent = value;
  	theInputRange.value = value;
    theRadialGauge.value = value;
		theRadialGaugeCSS.value = value;
  }
}
