onload = function() {

	// create the gauge
	var theGauge = new wijmo.gauge.RadialGauge('#theGauge', {
  	isReadOnly: false,
  	min: 0,
    max: 100,
    step: 5,
    value: 75,
    valueChanged: function(s, e) {
    	value.value = s.value;
    },
    showRanges: false,
    ranges: [
    	{ min: 0, max: 33, color: 'red' },
    	{ min: 33, max: 66, color: 'yellow' },
    	{ min: 66, max: 100, color: 'green' },
    ]
  });
  
  // change properties
  var value = new wijmo.input.InputNumber('#value', {
  	min: theGauge.min,
    max: theGauge.max,
    step: theGauge.step,
    value: theGauge.value,
    valueChanged: function(s, e) {
    	theGauge.value = s.value;
    }
  });
  var startAngle = new wijmo.input.InputNumber('#startAngle', {
    step: 10,
  	min: -360,
    max: 360,
    value: theGauge.startAngle,
    valueChanged: function(s, e) {
    	theGauge.startAngle = s.value;
    }
  });
  var sweepAngle = new wijmo.input.InputNumber('#sweepAngle', {
    step: 10,
  	min: -360,
    max: 360,
    value: theGauge.sweepAngle,
    valueChanged: function(s, e) {
    	theGauge.sweepAngle = s.value;
    }
  });
	document.getElementById('autoScale').addEventListener('click', function(e) {
  	theGauge.autoScale = e.target.checked;
  })
}
