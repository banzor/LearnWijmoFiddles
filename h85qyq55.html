onload = function() {

	// data used in the combo
	var theData = [
  	{ name: 'Zero', value: 0 },
  	{ name: 'Low', value: 10 },
  	{ name: 'Medium', value: 50 },
  	{ name: 'High', value: 100 },
  	{ name: 'Maximum', value: 150 },
  ];
  
  // show the data
  var theGrid = new wijmo.grid.FlexGrid('#theGrid', {
  	itemsSource: theData,
    isReadOnly: true,
    selectionMode: 'None'
  });
  
  // create the combo
  var theComboBox = new wijmo.input.ComboBox('#theComboBox', {
    displayMemberPath: 'name',
    selectedValuePath: 'value',
    selectedIndexChanged: function(s, e) {
    	updateText(s, 'displayMemberPath');
    	updateText(s, 'selectedValuePath');
    	updateText(s, 'selectedIndex');
    	updateText(s, 'selectedItem');
    	updateText(s, 'selectedValue');
    	updateText(s, 'text');
    },
  	itemsSource: theData
  });
  
  // show current values
  function updateText(combo, prop) {
		var value = combo[prop];
    if (wijmo.isObject(value)) {
  		value = JSON.stringify(value);
		}
  	document.getElementById(prop).textContent = value;
  }
}

  <p>
    The ComboBox control is one of the most powerful and flexible in Wijmo's input module. It can be used to edit strings and to select items from lists. It is used as a base class for several other controls including AutoComplete, MultiSelect, InputTime,
    and Menu.</p>
  <p>
    The main properties in the ComboBox control are:</p>
  <ul>
    <li>
      <b>itemsSource</b>
      Gets or sets an array (or CollectionView) containing the items
      (values or objects) to choose from.
    </li>
    <li>
      <b>displayMemberPath</b>
      Gets or sets the name of the property of the items to display
      in the input element and in the drop-down list.
    </li>
    <li>
      <b>selectedValuePath</b>
      Gets or sets the name of the property of the items to use
      when getting or setting the value of the <b>selectedValue</b>
      property.
    </li>
    <li>
      <b>selectedItem</b>
      Gets or sets the currently selected item in the
      <b>itemsSource</b> collection.
    </li>
    <li>
      <b>selectedIndex</b>
      Gets or sets the index of the currently selected item.
    </li>
    <li>
      <b>selectedValue</b>
      Gets or sets the value of the <b>selectedValuePath</b>
      property in the currently selected item.
    </li>
    <li>
      <b>isEditable</b>
      Gets or sets a value that determines whether users should 
      be allowed to enter values not present in the <b>itemsSource</b>.
    </li>
    <li>
      <b>isRequired</b>
      Gets or sets a value that determines whether users should be
      allowed clear the text in the control.
    </li>
  </ul>
