onload = function() {

	// create two DropDowns
	var theComboBox = new wijmo.input.ComboBox('#theComboBox', {
  	itemsSource: 'Ringo,George,John,Paul'.split(',')
  });
	var theInputDate = new wijmo.input.InputDate('#theInputDate');

	// customize the DropDowns
  document.getElementById('isAnimated').addEventListener('click', function(e) {
  	theComboBox.isAnimated = e.target.checked;
  	theInputDate.isAnimated = e.target.checked;
  });
  document.getElementById('isDroppedDown').addEventListener('click', function(e) {
  	theComboBox.isDroppedDown = e.target.checked;
  	//theInputDate.isDroppedDown = e.target.checked;
  });
  document.getElementById('dropDownCssClass').addEventListener('click', function(e) {
  	theComboBox.dropDownCssClass = e.target.checked ? 'custom-dd' : '';
  	theInputDate.dropDownCssClass = e.target.checked ? 'custom-dd' : '';
  });
}
<div class="container">

  <h1>
    DropDown
  </h1>
  <p>
    The DropDown control is an abstract class, used as a base
    for several controls. It is composed of the following elements:</p>
  <ul>
    <li>
      <b>input element</b> (exposed by the <b>inputElement</b> property):<br/>
      An HTML input element used to display and edit the current value.
      In a few cases, this element may be replaced with a non-editable
      element (e.g. Menu and MultiSelect controls).
    </li>
    <li>
      <b>drop-down button</b> 
      A button used to show or hide the drop-down element.
    </li>
    <li>
      <b>drop-down element (exposed by the <b>dropDown</b> property)</b>:
      An HTML element shown when the user presses the drop-down
      button or the F4 key. The user interacts with the drop-down
      element to update the value of the input element and of the
      control.
    </li>
  </ul>
  <p>
    All DropDown controls have the following properties and events:</p>
  <ul>
    <li>
      <b>text</b>
      Gets or sets the current value of the <b>inputElement</b>.
    </li>
    <li>
      <b>textChanged</b>
      Event that fires when the value of the <b>text</b> property changes..
    </li>
    <li>
      <b>isDroppedDown</b>
      Gets or sets a value that determines whether the drop-down is
      currently visible.
    </li>
    <li>
      <b>isDroppedDownChanging, isDroppedDownChanged</b>
      Events that fire when the value of the <b>isDroppedDown</b>
      property changes.
    </li>
    <li>
      <b>isAnimated</b>
      Property that determines whether the control should use 
      animations when showing the drop-down.
    </li>
    <li>
      <b>dropDownCssClass</b>
      Class added to the drop-down element in order to allow CSS 
      styling of the drop-down.<br/>
      This property is useful because when the drop-down is displayed,
      it is usually re-parented to become a direct child of the
      document's body, which prevents it from inheriting the styles 
      of the control's host element.
    </li>
  </ul>
  
  <p>
    Here is a simple example that demonstrates some of these
    properties:</p>
    
  <div class="row">
    <div class="col-xs-6">
      <label for="isAnimated">isAnimated: </label>

      <input id="isDroppedDown" type="checkbox" value="false">

      <input id="dropDownCssClass" type="checkbox" value="false">

    </div>
