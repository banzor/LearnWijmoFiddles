onload = function() {

  // initialize pivot engine
  var ng = new wijmo.olap.PivotEngine({
    itemsSource: getData(10000), // raw data
    showRowTotals: 'Subtotals',
    valueFields: ['Amount'], // summarize amounts
    rowFields: ['Buyer', 'Type'] // by buyer and by type
  });
  ng.fields.getField('Amount').format = 'c0'; // customize field

  // show raw data
  var rawData = new wijmo.grid.FlexGrid('#rawData', {
    itemsSource: ng.collectionView
  });

  // show summary
  var pivotGrid = new wijmo.olap.PivotGrid('#pivotGrid', {
    itemsSource: ng
  });

  // get the raw data
  function getData(cnt) {
    var today = new Date(),
    		buyers = 'Mom,Dad,Kelly,Sheldon'.split(','),
        types = 'Food,Clothes,Fuel,Books,Sports,Music'.split(','),
    		data = [];
    for (var i = 0; i < cnt; i++) {
    	data.push({
      	date: wijmo.DateTime.addDays(today, -Math.random() * 365),
        buyer: randomItem(buyers),
        type: randomItem(types),
        amount: 20 + Math.random() * 1000
      });
    }
    return data;
  }
  function randomItem(arr) {
  	return arr[Math.floor(Math.random() * arr.length)];
  }

}

    </li>
    <li>
      Build view definitions by configuring each field and copying
      it to one of the output field lists (<b>rowFields</b>,
      <b>columnFields</b>, <b>valueFields</b>, and <b>filterFields</b>).
    </li>
    <li>
      If you want to allow users to customize the view definitions
      using drag/drop, bind the engine to a <b>PivotPanel</b> control.
    </li>
    <li>
      Get the summary view from the <b>pivotView</b> property, or
      bind the engine to <b>PivotGrid</b> or <b>PivotChart</b> 
      controls.
    </li>
  </ol>
