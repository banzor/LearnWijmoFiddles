onload = function() {

	// create several InputNumber controls and assign them formats
  var parent = document.getElementById('inputNumbers');
  var formats = 'n0,n2,c0,c2,p0,p2'.split(',');
	var template = '<div>' +
  	'<label for={id}>Step {step}: </label>' +
    '<input id={id}>' +
    '</div>';
  for (var i = 0; i < formats.length; i++) {
  	var fmt = formats[i],
    		id = 'theInputNumberF' + fmt,
        step = fmt[0] == 'p' ? .1 : (i + 1);
  	var html = wijmo.format(template, {
    	fmt: fmt,
      id: id,
      step: step
		});
  	parent.appendChild(wijmo.createElement(html));
  	var theNumber = new wijmo.input.InputNumber('#' + id, {
  		format: fmt,
      step: step
    });
  }
}
