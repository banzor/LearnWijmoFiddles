onload = function() {

  // create the chart
  var theChart = new wijmo.chart.FlexChart('#theChart', {
    itemsSource: getData(),
    bindingX: 'country'
  });
  
  // auto-generate series
  var item = theChart.itemsSource[0];
  for (var k in item) {
  	if (wijmo.isNumber(item[k])) {
    	var series = new wijmo.chart.Series();
      series.binding = k;
			series.name = wijmo.toHeaderCase(k);
      series['visible'] = true; // add 'visible' property for binding
    	theChart.series.push(series);
    }
  }
  
  // show the chart's series in a grid
  var theGrid = new wijmo.grid.FlexGrid('#theGrid', {
  	autoGenerateColumns: false,
    columns: [
    	{ binding: 'binding', header: 'Binding', isReadOnly: true },
    	{ binding: 'name', header: 'Name', isRequired: true },
      { binding: 'visibility', header: 'Visibility', dataMap: getDataMap(wijmo.chart.SeriesVisibility) },
      { binding: 'chartType', header: 'Chart Type', dataMap: getDataMap(wijmo.chart.ChartType), isRequired: false }
    ],
  	itemsSource: theChart.series,
    cellEditEnded: function() {
    	theChart.refresh(true);
    }
  });
  
  // build a DataMap for a given enum
  function getDataMap(enumClass) {
    var pairs = [];
    for (var key in enumClass) {
        var val = parseInt(key);
        if (!isNaN(val)) {
        	pairs.push({ key: val, name: enumClass[val] });
        }
    }
    return new wijmo.grid.DataMap(pairs, 'key', 'name');
	}

	// generate some random data
  function getData() {
    var countries = 'US,Canada,Mexico,Japan,Korea,China,Germany,France,UK,Italy,Greece'.split(','),
    	data = [];
	  for (var i = 0; i < countries.length; i++) {
    	data.push({
      	country: countries[i],
        leads: Math.random() * 100000,
        qualified: Math.random() * 50000,
        analysis: Math.random() * 5000,
        quotes: Math.random() * 4000,
        negotiations: Math.random() * 1000,
      	sales: Math.random() * 800
			});
    }
  	return data;
  }
}
