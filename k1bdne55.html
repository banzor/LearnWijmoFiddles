onload = function() {

  // create the tree
  var theTree = new wijmo.nav.TreeView('#theTree', {
    itemsSource: getData(),
    displayMemberPath: 'header',
    childItemsPath: 'items',
  });
  theTree.selectedItem = theTree.itemsSource[0];

	// refresh the node with new data
  document.getElementById('btnItemData').addEventListener('click', function () {
  	theTree.selectedNode.refresh({
    	header: 'given itemData ' + Date.now(), items: [
      	{ header: 'first child' },
        { header: 'second child' },
			]
		});
	});

  // update the data, refresh the node
  document.getElementById('btnItemsSource').addEventListener('click', function () {
  	var node = theTree.selectedNode,
    		arr = node.itemsSource;
		arr[node.index] = {
			header: 'updated itemData ' + Date.now(), items: [
				{ header: 'first child' },
				{ header: 'second child' },
			]
		};
		node.refresh();
	});

  // re-load the whole tree
  document.getElementById('btnReload').addEventListener('click', function () {
  	theTree.loadTree(true);
	});
  
  // create some data
  function getData() {
    return [
    	{ header: 'Parent 1', items: [
      	{ header: 'Child 1.1' },
        { header: 'Child 1.2' },
        { header: 'Child 1.3' }]
    	},
      { header: 'Parent 2', items: [
      	{ header: 'Child 2.1' },
        { header: 'Child 2.2' }]
    	},
      { header: 'Parent 3', items: [
      	{ header: 'Child 3.1' }]
    	}
		];
  }
}
