onload = function() {

  var theTabPanel = new wijmo.nav.TabPanel('#theTabPanel');

  // toggle autoSwitch property
  document.getElementById('autoSwitch').addEventListener('click', function(e) {
    theTabPanel.autoSwitch = e.target.checked;
  });
}
