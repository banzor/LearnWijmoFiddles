onload = function() {

	// create the chart
  var theChart = new wijmo.chart.FlexChart('#theChart', {
  	itemsSource: getData(),
    chartType: 'Area',
    stacking: 'Stacked',
    bindingX: 'country',
    series: [
      { 
      	binding: 'expenses', 
        name: 'Expenses',
        style: {
					fill: 'l(0, 0, 0, 1)#ff0000-#ff0000:1:0.5', 
          stroke:'darkred',
          strokeWidth: 1        
        }
			},
    	{
      	binding: 'sales',
        name: 'Sales',
        style: {
      		fill:'l(0, 0, 0, 1)#00b050-#00b050:1:0.5', 
        	stroke:'darkgreen',
        	strokeWidth: 1
      	}
			}
    ]
  });
  
	// create some random data
  function getData() {
	  var countries = 'US,Canada,Mexico,Germany,Austria,Belgium,Denmark,France,Russia,Finland,UK,Italy,Japan,Korea,China'.split(','),
	    data = [];
	  for (var i = 0; i < countries.length; i++) {
	    data.push({
	      country: countries[i],
	      sales: Math.random() * 10000,
	      expenses: Math.random() * 5000,
	      downloads: Math.round(Math.random() * 20000),
	    });
	  }
  	return data;
  }

}

  <p>
    The FlexChart allows you to specify gradients for series style
    properties. The gradients are specified as strings with 
    the following syntax:</p>
    
  <pre>{type}({coords}){color[:offset[:opacity]]}</pre>
  
  <p>
    Where:</p>
  <ol>
    <li>
      <b>{type}</b> is either 'l' or 'r', for linear or radial gradients,</li>
    <li>
      <b>{coords}</b> is a set of three or four numbers that specify the
      gradient direction (x1/y1/x2/y2 for linear, cx.cy.r for radial), and</li>
    <li>
      <b>{color:offset:opacity}</b> is an HTML color string optionally 
      followed by offset and opacity values. You may add add additional colors
      using a hyphen as a separator.</li>      
  </ol>
