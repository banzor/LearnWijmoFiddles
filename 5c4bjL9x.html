onload = function() {

	// create the gauge
  var theGauge = new wijmo.gauge.RadialGauge('#theGauge', {
  	isReadOnly: false,
    hasShadow: false,
    showText: 'All',
    startAngle: -30,
    sweepAngle: 240,
    value: 50,
  });
  theGauge.pointer.thickness = .15;
  theGauge.thumbSize = 20;

	var theColor = new wijmo.input.InputColor('#theColor', {
  	value: theGauge.face.color,
		valueChanged: function(s, e) {
			theGauge.pointer.color = s.value;
		}	
	});
	var theThumbSize = new wijmo.input.InputNumber('#theThumbSize', {
  	min: 0,
  	max: 50,
    step: 5,
    format: 'n0',
    value: theGauge.thumbSize,
		valueChanged: function(s, e) {
			theGauge.thumbSize = s.value;
		}	
	});

}
