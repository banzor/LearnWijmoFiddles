onload = function() {

	// InputNumber with a restricted range
	var theInputNumber = new wijmo.input.InputNumber('#theInputNumber', {
  		format: 'n0',
      value: 1,
      min: 1,
      max: 5,
      step: 1
	});
}

<div class="container">

  <h1>
    InputNumber Ranges
  </h1>

  <p>
      Use the InputNumber's <b>min</b> and <b>max</b> properties
      to restrict the range of values that can be entered.</p>  
  <p>
      For example, the InputNumber below only accepts values 
      between one and five:</p>
  <label for="theInputNumber">Guests</label>
  <div id="theInputNumber"></div>
  <p>
    Note that the control will not validate the range
    as the value is being edited, but will clamp the
    value to the valid range when the control loses focus.</p>
  <p>
    For example, you can type "123" in the control above, 
    but as soon as you move the focus to another element
    the control will apply the constraints and change the
    value to five, the value of the <b>max</b> property:</p>
</div>
