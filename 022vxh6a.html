onload = function() {

  // create some random data
  var countries = 'US,Germany,UK,Japan,Italy,Greece,Spain,Portugal,Australia'.split(',');
  var data = [];
  for (var i = 0; i < countries.length; i++) {
    data.push({
      country: countries[i],
      downloads: Math.round(Math.random() * 20000),
      sales: Math.random() * 10000,
      expenses: Math.random() * 5000
    });
  }

  // create a CollectionView
  var view = new wijmo.collections.CollectionView(data, {
		sortDescriptions: ['country']    
  });

  // create grid for editing
  var theGrid = new wijmo.grid.FlexGrid('#theGrid', {
    itemsSource: view
  });
}

<div class="container">

  <h1>
    CollectionView Editing
  </h1>
  <p>
    The <b>CollectionView</b> provides support for editing items
    with methods similar to those found in .NET's
    <b>IEditableCollectionView</b> interface:
    <b>editItem</b>, <b>commitEdit</b>, and <b>cancelEdit</b>.</p>
  <p>
    The <b>editItem</b> method saves a copy of the item and
    puts the collection in 'edit mode'. While in edit mode,
    the view is not refreshed, so items will not be sorted
    or filtered out of view during the edit process.</p>
  <p>
    The <b>commitEdit</b> method exits edit mode so sorting
    and filtering become active again. If the item has changed,
    the <b>collectionChanged</b> event fires so bound controls
    can show the changes.</p>
  <p>
    The <b>cancelEdit</b> method restores the original data
    and exits edit mode.</p>
  <p>
    You can see how this works by editing the grid below.
    When editing starts, a pencil icon appears on the 
    row header to indicate the collection is in edit
    mode.</p>
  <p>
    If you press the Escape key while in edit mode, the
    edits will be canceled and the original data will be
    restored.</p>
  <p>
    If you move the selection to a different row, or move
    the focus away from the grid, the edits will be committed.
    At this point, the collection will refresh and any
    active filtering/sorting will be applied using the
    new item values.</p>

</div>
