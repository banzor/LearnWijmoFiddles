onload = function() {

	// create the controls
  var ctls = [];
  ctls.push(new wijmo.input.InputDate('#theInputDate', {
  	isRequired: false
  }));
  ctls.push(new wijmo.input.InputTime('#theInputTime', {
  	isRequired: false
  }));
  ctls.push(new wijmo.input.InputDateTime('#theInputDateTime', {
  	isRequired: false
  }));
  ctls.push(new wijmo.input.InputNumber('#theInputNumber', {
  	isRequired: false,
    step: 1,
    value: 1234,
    format: 'n2'
  }));
  ctls.push(new wijmo.input.ComboBox('#theComboBox', {
  	isRequired: false,
    itemsSource: 'apples,oranges'.split(',')
  }));

	// add clear button to all the controls
  ctls.forEach(function(ctl) {
  	addClearButton(ctl)	
  });
  
	// add clear button to a Wijmo input control
	function addClearButton(ctl) {
  	var host = ctl.hostElement,
    		input = ctl.inputElement;
		host.classList.add('wj-clear-input');
    host.addEventListener('mousedown', function (e) {
    	if (e.offsetX < 0 && wijmo.hasClass(e.target, 'wj-input-group-btn') &&  !wijmo.closest(e.target, '.wj-state-empty')) {
      	e.preventDefault();
        e.stopImmediatePropagation();
        input.value = '';
        input.dispatchEvent(inputEvent);
        input.focus();
			}
		}, true);
	}
  var inputEvent = document.createEvent('HTMLEvents');
  inputEvent.initEvent('input', true, false);
  
}
