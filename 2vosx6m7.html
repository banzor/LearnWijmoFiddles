onload = function() {

  // create some random data
  var countries = 'US,Germany,UK,Japan,Italy,Greece,China,Spain,Sweden,Norway,Denmark,Finland'.split(','),
    data = [];
  for (var i = 0; i < countries.length; i++) {
    data.push({
      id: i,
      country: countries[i],
      active: i % 5 != 0,
      downloads: Math.round(Math.random() * 200000),
      sales: Math.random() * 100000,
      expenses: Math.random() * 50000
    });
  }

  // bind two grids to the same array
  // each shows an independent view of the same data
  var gridArray1 = new wijmo.grid.FlexGrid('#gridArray1', {
    itemsSource: data
  });
  var gridArray2 = new wijmo.grid.FlexGrid('#gridArray2', {
    itemsSource: data
  });
  
  // bind two grids to the same CollectionView
  // both share the same view (sorting, filtering, selection, etc)
  var view = new wijmo.collections.CollectionView(data);
  var gridCv1 = new wijmo.grid.FlexGrid('#gridCv1', {
    itemsSource: view
  });
  var gridCv2 = new wijmo.grid.FlexGrid('#gridCv2', {
    itemsSource: view
  });
  
  // bind a ListBox to the same view
  var listBox = new wijmo.input.ListBox('#listBox', {
  	itemsSource: view,
    displayMemberPath: 'country'
  })
  
  // add filters to all grids on the page
  var grids = document.querySelectorAll('.wj-flexgrid');
  for (var i = 0; i < grids.length; i++) {
  	var grid = wijmo.Control.getControl(grids[i]);
    var filter = new wijmo.grid.filter.FlexGridFilter(grid);
  }
}

<div class="container">

  <h1>
    FlexGrid Architecture
  </h1>
  <p>
    The FlexGrid displays data in tabular form and allows
    users to edit it or to modify the views by sorting, 
    filtering, and grouping it.</p>
  <p>
    Like all Wijmo controls, the grid delegates all 
    data-related tasks to the 
    <a href="https://wijmo.com/5/docs/topic/wijmo.collections.CollectionView.Class.html" target="_blank">CollectionView</a>
    class.</p>
  <p>
    When you set the grid's <b>itemsSource</b> property to an array,
    or 
    <a href="https://wijmo.com/5/docs/topic/wijmo.collections.ObservableArray.Class.html" target="_blank">ObservableArray</a>,
    the grid automatically creates a CollectionView instance 
    to wrap the original array. This CollectionView is exposed 
    by the <b>collectionView</b> property in case you want to 
    access it.</p>
  <p>
    For example, the grids below are bound to the same data array.
    Each has its own internal CollectionView, so they can be sorted,
    filtered, and grouped independently. The current item
    (associated with the grid selection) is also independent:</p>
  <div class="row">
    <div class="col-xs-6">
      <div id="gridArray1"></div>
    </div>
    <div class="col-xs-6">
      <div id="gridArray2"></div>
    </div>
  </div>

  <p>
    By contrast, the two grids below are bound to the same CollectionView.
    Because of this, they show the same view. Sorting, filtering, or
    selecting on one grid affects the other.</p>
  <div class="row">
    <div class="col-xs-6" >
      <div id="gridCv1"></div>
    </div>
    <div class="col-xs-6">
      <div id="gridCv2"></div>
    </div>
  </div>
  
  <p>
    Notice how the sorting, filtering, and selection applied
    to the common view are also reflected on other Wijmo 
    controls such as the <b>ListBox</b>:</p>
  <div id="listBox"></div>


</div>
