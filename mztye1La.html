onload = function() {

	// create the tree
  var tree = new wijmo.nav.TreeView('#theTree', {
  	itemsSource: getData(),
		displayMemberPath: 'header',
    childItemsPath: 'items',
    isReadOnly: false,
		nodeEditStarting: function (s, e) {
    	if (e.node.hasChildren) {
				e.cancel = true;
			}    
		}
	});
  
  // get the tree data
  function getData() {
		return [
    	{ header: 'Electronics', items: [
				{ header: 'Trimmers/Shavers' },
