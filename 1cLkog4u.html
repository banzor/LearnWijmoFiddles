onload = function() {

	theSSN = new wijmo.input.InputMask('#theSSN', {
  	mask: '000-00-0000',
    valueChanged: showRaw
  });
	theZip = new wijmo.input.InputMask('#theZip', {
  	mask: '00000',
    valueChanged: showRaw
  });
	theZip4 = new wijmo.input.InputMask('#theZip4', {
  	mask: '00000-0000',
    valueChanged: showRaw
  });
	thePhone = new wijmo.input.InputMask('#thePhone', {
  	mask: '(999) 000-0000',
    valueChanged: showRaw
  });
  
  function showRaw(s, e) {
  	var el = document.getElementById(s.hostElement.id + 'Raw');
    el.textContent = ' rawValue: ' + s.rawValue;
  }
}
