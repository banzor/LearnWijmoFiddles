onload = function() {

	// create the chart
  var theChart = new wijmo.chart.FlexChart('#theChart', {
    axisY: {
    	format: 'n0,',
      title: 'US$ (thousands)'
    },
  	itemsSource: getData(),
    bindingX: 'country',
    series: [
    	{ binding: 'sales', name: 'Sales' },
      { binding: 'expenses', name: 'Expenses' },
    ]
  });

	// change the label angle
  var labelAngle = new wijmo.input.ComboBox('#labelAngle', {
    itemsSource: [ 0, 30, 45, 60, 90 ],
    placeholder: 'auto',
    isRequired: false,
    text: null,
    textChanged: function(s, e) {
    	theChart.axisX.labelAngle = s.selectedValue;
    },
    text: ''
  })

	// create some random data
  function getData() {
	  var countries = 'US,Canada,Mexico,Germany,UK,France,Italy,Japan,Korea,China'.split(','),
	    data = [];
	  for (var i = 0; i < countries.length; i++) {
	    data.push({
	      country: countries[i],
	      sales: 1000 + Math.random() * 10000,
	      expenses: 1000 + Math.random() * 5000,
	      downloads: Math.round(Math.random() * 20000),
	    });
	  }
  	return data;
  }

}
