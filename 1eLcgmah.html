onload = function() {

	// create the gauge
  var theGauge = new wijmo.gauge.RadialGauge('#theGauge', {
  	isReadOnly: false,
    hasShadow: false,
    showText: 'All',
    showTicks: true,
    tickSpacing: 10,
    startAngle: -30,
    sweepAngle: 240,
    value: 50,
  });
  theGauge.pointer.thickness = .5;

	// customize tickmarks
	document.getElementById('showTicks').addEventListener('click', function(e) {
 		theGauge.showTicks = e.target.checked; 
  });
	var tickSpacing = new wijmo.input.InputNumber('#tickSpacing', {
  	min: 0,
  	max: 100,
    step: 5,
    format: 'n0',
    value: theGauge.tickSpacing,
		valueChanged: function(s, e) {
			theGauge.tickSpacing = s.value;
		}	
	});
  var tickColor = new wijmo.input.InputColor('#tickColor', {
    value: 'white',
		valueChanged: function(s, e) {
			var style = theGauge.hostElement.querySelector('.wj-ticks').style;
      style.stroke = s.value;
		}	
	});
  var tickWidth = new wijmo.input.InputNumber('#tickWidth', {
  	min: 0,
  	max: 5,
    step: .25,
    format: 'n1',
    value: 2,
		valueChanged: function(s, e) {
			var style = theGauge.hostElement.querySelector('.wj-ticks').style;
      style.strokeWidth = s.value;
		}	
	});

}
