onload = function() {

  var countries = new wijmo.collections.ObservableArray('US,Germany,UK,Japan,Italy,Greece'.split(','));

  new wijmo.input.ComboBox('#theCombo', {
    itemsSource: countries,
    isEditable: true,
    lostFocus: lostFocus
  });

  new wijmo.input.AutoComplete('#theAutoComplete', {
    itemsSource: countries,
    lostFocus: lostFocus
  });

  new wijmo.input.ListBox('#theList', {
    itemsSource: countries
  });

  // add item to the list when a control loses focus
  function lostFocus(s, e) {
    var item = s.text; //wijmo.toHeaderCase(s.text);
    if (s.text && countries.indexOf(item) < 0) {
      console.log('adding "' + item + '" to the list...');
      countries.push(item);
    }
  }

}
