onload = function() {

  // create the combo
  var theCombo = new wijmo.input.ComboBox('#theCombo', {
		itemsSource: getOptions(),
    displayMemberPath: 'header',
    selectedIndex: 1,
    formatItem: formatListItem
	});
  
  // create the AutoComplete
  var theAutoComplete = new wijmo.input.AutoComplete('#theAutoComplete', {
		itemsSource: getOptions(),
    displayMemberPath: 'header',
    selectedIndex: 1,
    formatItem: formatListItem
	});

	// format list items to format and disable category names
  function formatListItem(s, e) {
		var dataItem = e.data;
    if (dataItem.category) {
    	e.item.classList.add('category');
      e.item.classList.add('wj-state-disabled');
      e.item.disabled = true;
		}
  }
  
  // get categorized list of options for the combo
	function getOptions() {
  	return [
    	{ header: 'Colors', category: true },
      	{ header: 'Red' },
        { header: 'Green' },
        { header: 'Blue' },
			{ header: 'Flavors', category: true },
      	{ header: 'Chocolate' },
        { header: 'Vanilla' },
        { header: 'Strawberry' }
		]
	}
}
