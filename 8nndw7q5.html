onload = function() {

  // create the chart
  var theChart = new wijmo.chart.FlexChart('#theChart', {
    itemsSource: getData(),
    bindingX: 'country',
    chartType: 'Bar',
    axisY: {
    	reversed: true
    },
    series: [
    	{ binding: 'sales', name: 'Sales' },
      { binding: 'expenses', name: 'Expenses' }
		]
  });

	// reverse Y axis
  document.getElementById('reverseY').addEventListener('click', function(e) {
  	theChart.axisY.reversed = e.target.checked;
  });
  
  // generate some random data
  function getData() {
    var countries = 'US,Canada,Mexico,UK,Germany,France,Japan,Korea,China'.split(','),
      data = [];
    for (var i = 0; i < countries.length; i++) {
      data.push({
        country: countries[i],
        downloads: Math.round(Math.random() * 20000),
        sales: Math.random() * 10000,
        expenses: Math.random() * 5000
      });
    }
    return data;
  }
}
