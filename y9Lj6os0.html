onload = function() {

  // create some random data
  var countries = 'US,Germany,UK,Japan,Sweden,Norway,Denmark'.split(','),
    data = [];
  for (var i = 0; i < countries.length; i++) {
    data.push({
      id: i,
      country: countries[i],
      active: i % 5 != 0,
      downloads: Math.round(Math.random() * 200000),
      sales: Math.random() * 100000,
      expenses: Math.random() * 50000
    });
  }

	// auto-generated columns
  var theGridAutoGen = new wijmo.grid.FlexGrid('#theGridAutoGen', {
    showAlternatingRows: false,
  	itemsSource: data
	});

  // explicit columns 
  var theGridExplicitCols = new wijmo.grid.FlexGrid('#theGridExplicitCols', {
    showAlternatingRows: false,
    autoGenerateColumns: false,
    columns: [
    	{ binding: 'country', header: 'Country', width: '*'},
    	{ binding: 'downloads', header: 'Downloads (k)', format: 'n0,' }, // thousands
    	{ binding: 'sales', header: 'Sales', format: 'n2' }, 
    	{ binding: 'expenses', header: 'Expenses', format: 'n2' },
    	{ binding: 'active', header: 'Active'}
    ],
    itemsSource: data,
  });
}

  <p>

    the grid will scan the data objects and automatically
    generate columns for each property in the data items.
    For example:</p>

  </div>
  
  <p>
    Auto-generating columns works well for simple data sources
    or while testing/developing applications. But in most scenarios,
    you will want to turn off the automatic column generation and
    specify the columns in code. This way you get complete control
    over which columns are displayed, in what order, their widths, 
    format, headers, etc.<p>
  <p>
    Each column has a <b>binding</b> property that determines which
    property of the data item that should be displayed in the cell.
