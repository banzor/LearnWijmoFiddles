onload = function() {

    // generate some random data
    var countries = 'US,Germany,UK,Japan,Italy,Greece'.split(','),
      data = [];
    for (var i = 0; i < countries.length; i++) {
      data.push({
        country: countries[i],
        downloads: Math.round(Math.random() * 20000),
        sales: Math.random() * 10000,
        expenses: Math.random() * 5000
      });
    }

    // create grid and show data
    var grid = new wijmo.grid.FlexGrid('#theGrid', {
      itemsSource: data,
      showMarquee: true,
      showSelectedHeaders: 'All',
      showAlternatingRows: false
    });

    // toggle grid's background
    document.getElementById('toggleBackground').addEventListener('click', function() {

      // get control being hosted by element #theGrid
      var ctl = wijmo.Control.getControl('#theGrid');

      // get host element from 'grid' control
      var host = ctl.hostElement;

      // toggle font weight
      var style = host.style;
			style.fontWeight = style.fontWeight ? '' : 'bold';
    });

    // resize the grid the WRONG way
    document.getElementById('resizeBad').addEventListener('click', function() {
      var style = grid.hostElement.style;
      style.height = style.height ? '' : '100px';
    });

    // resize the grid the RIGHT way
    document.getElementById('resizeGood').addEventListener('click', function() {
      var style = grid.hostElement.style;
      style.height = style.height ? '' : '100px';
      grid.invalidate(); // update control layout after resizing the element
      
      // or (if you have many controls)
      //wijmo.Control.invalidateAll();
    });

}

  <h1>
    Controls and Elements</h1>
  <p>
    Every Wijmo 5 control is associated with an HTML element that
    hosts it on the page.</p>

  <p>
    Use the <b>hostElement</b> property (available on every Wijmo
    control) to get a reference to the element that is hosting the 
    control.</p>
  <p>
    Use the static method <b>wijmo.Control.getControl(element)</b>
    to get a reference to the control being hosted by an element.</p>

  <p>
    When the user resizes the browser window, Wijmo automatically
    updates the layout of all controls based on the size of their 
    host elements.</p>
  <p>
    If the size or visiblility of the host element is changed in
    code, however, the control will not be automatically notified,
    and its layout may become invalid. This often happens in 
    applications that have tab controls or dynamic views.</p>
  <p>
    In these cases, you should call the
    <a href="https://wijmo.com/5/docs/topic/wijmo.Control.Class.html#invalidateAll">
    wijmo.Control.invalidateAll</a> method to notify the controls
    that their host element has changed and they should update their
    layout.</p>
  <p>
    For example, use the buttons below to resize a grid with and 
    without calls to <b>invalidateAll</b>:
  </p>

  <button id="toggleBackground" class="btn btn-primary">Toggle Font</button>
  <button id="resizeBad" class="btn btn-danger">Toggle Size (WRONG)</button>
  <button id="resizeGood" class="btn btn-success">Toggle Size (RIGHT)</button>
  <div id="theGrid"></div>

</div>
