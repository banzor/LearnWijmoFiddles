onload = function() {

	// create the gauges
  var linear = new wijmo.gauge.LinearGauge('#linear', {
  	isReadOnly: false,
    hasShadow: false,
    thumbSize: 12,
    value: 50
  });
  linear.face.thickness = .25;
  linear.pointer.thickness = .25;
  
  var radial = new wijmo.gauge.RadialGauge('#radial', {
  	isReadOnly: false,
    hasShadow: false,
    showText: 'Value',
    thumbSize: 6,
    value: 50
  });
  radial.face.thickness = .05;
  radial.pointer.thickness = .05;
  

}
