onload = function() {

  // create some random data
  var countries = 'US,Germany,UK,Japan,Italy,Greece'.split(',');
  var data = [];
  for (var i = 0; i < 100; i++) {
    data.push({
    	id: i,
      country: countries[i % countries.length],
      sales: Math.random() * 10000,
      expenses: Math.random() * 5000
    });
  }

  // show the data in a grid with sticky headers
  var theGridSticky = new wijmo.grid.FlexGrid('#theGridSticky', {
    itemsSource: data,
    stickyHeaders: true
  });

  // and in a grid without sticky headers
  var theGridNormal = new wijmo.grid.FlexGrid('#theGridNormal', {
    itemsSource: data
  });

}
